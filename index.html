  <script>

    // THEME TOGGLE

    const themeBtn = document.getElementById('themeBtn');

    const soundBtn = document.getElementById('soundBtn');

    const audio = document.getElementById('bgAudio');

    const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;

    const saved = localStorage.getItem('theme');

    if(saved){ document.documentElement.dataset.theme = saved; }

    else if(!prefersDark){ document.documentElement.dataset.theme = 'light'; }

    themeBtn.addEventListener('click', ()=>{

      const t = document.documentElement.dataset.theme === 'light' ? '' : 'light';

      document.documentElement.dataset.theme = t;

      if(t) localStorage.setItem('theme', t); else localStorage.removeItem('theme');

    });


    // SOUND TOGGLE

    let playing=false; soundBtn.addEventListener('click', async ()=>{

      try{ if(!playing){ await audio.play(); soundBtn.textContent='❚❚ Sound'; playing=true; } else { audio.pause(); soundBtn.textContent='▶︎ Sound'; playing=false; } }

      catch(e){ console.log(e); }

    });


    // KEYBOARD SHORTCUTS (M / F / J)

    const jump = id => document.getElementById(id).scrollIntoView({behavior:'smooth'});

    window.addEventListener('keydown', e=>{

      if(['INPUT','TEXTAREA'].includes(document.activeElement.tagName)) return;

      if(e.key.toLowerCase()==='m') jump('music');

      if(e.key.toLowerCase()==='f') jump('fashion');

      if(e.key.toLowerCase()==='j') jump('journal');

      if(e.key.toLowerCase()==='t') jump('top');

    });


    // CUSTOM CURSOR DOT

    const c = document.getElementById('cursor');

    window.addEventListener('pointermove', e=>{ c.style.transform = `translate(${e.clientX-7}px, ${e.clientY-7}px)`; });


    // DRAGGABLE STICKER

    (function(){

      const el = document.getElementById('sticker');

      let offX=0, offY=0, dragging=false;

      el.addEventListener('pointerdown', (e)=>{ dragging=true; el.setPointerCapture(e.pointerId); offX=e.clientX - el.offsetLeft; offY=e.clientY - el.offsetTop; });

      window.addEventListener('pointermove', (e)=>{ if(!dragging) return; el.style.left = (e.clientX - offX) + 'px'; el.style.top = (e.clientY - offY) + 'px'; });

      window.addEventListener('pointerup', ()=> dragging=false);

    })();

  </script>
